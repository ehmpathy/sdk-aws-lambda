## subcomponent contracts

### genLambdaEndpoint

given(genLambdaEndpoint contract)
  then(exposes: genLambdaEndpoint(config: { schema: { input, output }, invoke }) => Handler)
  then(Handler type is compatible with AWS Lambda handler signature)
  then(validates input against schema.input before invoke)
  then(validates output against schema.output after invoke)
  then(wraps invoke errors with context)
  then(injects trail-aware context.log into invoke)
  then(throws ValidationError for schema failures)

### genLambdaEndpoint.forApiGateway

given(genLambdaEndpoint.forApiGateway contract)
  then(exposes: genLambdaEndpoint.forApiGateway(config: { schema, invoke, cors }) => APIGatewayHandler)
  then(APIGatewayHandler type is compatible with AWS API Gateway handler signature)
  then(parses JSON body from API Gateway event)
  then(returns API Gateway response shape { statusCode, headers, body })
  then(includes CORS headers when origin matches)
  then(includes security headers on all responses)
  then(returns 400 for validation errors)
  then(returns 500 for invoke errors)
  then(sanitizes error responses (no stack traces))

### askLambdaEndpoint

given(askLambdaEndpoint contract)
  then(exposes: askLambdaEndpoint<TRes, TReq>(config: { which, event }, context?) => Promise<TRes>)
  then(constructs function name from which.service + which.function + stage)
  then(derives stage from context.env.STAGE)
  then(extracts trail.exid from context.log if present)
  then(injects trail into lambda payload)
  then(invokes AWS Lambda via @aws-sdk/client-lambda)
  then(parses response payload as JSON)
  then(throws LambdaInvocationError for lambda errors)
  then(integrates with-simple-cache for dedup)

### error classes

given(ValidationError contract)
  then(extends Error)
  then(includes field-level error details)
  then(distinguishes input vs output validation)

given(LambdaInvocationError contract)
  then(extends Error)
  then(includes target service/function)
  then(includes original error message)
  then(includes trail.exid for trace)

## composition boundaries

### handler middleware chain

given(standard handler composition)
  then(composes: trailMiddleware → validationMiddleware → invokeWrapper → outputValidation)
  then(trailMiddleware extracts or generates trail, injects into context)
  then(validationMiddleware validates input, short-circuits on failure)
  then(invokeWrapper calls user's invoke, catches errors)
  then(outputValidation validates return value)

### api gateway middleware chain

given(api gateway handler composition)
  then(extends standard handler composition)
  then(adds: bodyParser → corsMiddleware → securityHeaders → responseFormatter)
  then(bodyParser extracts JSON from event.body)
  then(corsMiddleware adds cors headers based on config)
  then(securityHeaders adds security headers)
  then(responseFormatter shapes response to API Gateway format)

### client composition

given(askLambdaEndpoint composition)
  then(composes: trailExtract → cacheCheck → lambdaInvoke → responseParser)
  then(trailExtract gets trail from context.log or generates)
  then(cacheCheck returns cached response if hit)
  then(lambdaInvoke calls AWS Lambda.invoke())
  then(responseParser parses JSON, detects errors, caches success)

## integration boundaries

### sdk-logs integration

given(trail integration)
  then(uses LogTrail interface from sdk-logs)
  then(uses genContextLogTrail for log creation)
  then(context.log.trail exposes current trail state)

### aws-sdk integration

given(lambda client integration)
  then(uses @aws-sdk/client-lambda (v3))
  then(Lambda.invoke() for invocation)
  then(handles Payload as Buffer, parses JSON)

### middy integration

given(middleware integration)
  then(uses @middy/core (v7) for middleware chain)
  then(custom middleware for validation, trail, cors)

### with-simple-cache integration

given(cache integration)
  then(uses with-simple-cache for dedup)
  then(cache key from service + function + payload hash)
